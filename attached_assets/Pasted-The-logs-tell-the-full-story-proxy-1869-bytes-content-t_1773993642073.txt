import axios from 'axios';
app.get('/stream', async (req, res) => {
  const q = req.query.q || req.query.url;
  const type = (req.query.type || 'mp3').toLowerCase();
  if (!q) return res.status(400).json({ error: 'Missing q or url param' });
  try {
    // Step 1: Get the CDN download URL from your own API (same server = same IP)
    const apiPath = type === 'mp4' ? 'mp4' : 'mp3';
    const paramKey = /^https?:\/\//i.test(q) ? 'url' : 'q';
    const apiRes = await axios.get(`https://apis.xwolf.space/download/${apiPath}`, {
      params: { [paramKey]: q, botkey: 'wolfbot_bypass_2024' },
      timeout: 30000
    });
    const downloadUrl = apiRes.data?.downloadUrl || apiRes.data?.download_url;
    if (!downloadUrl) return res.status(500).json({ error: 'No download URL from API' });
    // Step 2: Fetch and stream the file (same server IP as above = CDN allows it)
    const fileRes = await axios.get(downloadUrl, {
      responseType: 'stream',
      headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)', 'Accept': '*/*' },
      maxRedirects: 10,
      timeout: 120000
    });
    res.setHeader('Content-Type', fileRes.headers['content-type'] || (type === 'mp4' ? 'video/mp4' : 'audio/mpeg'));
    if (fileRes.headers['content-length']) res.setHeader('Content-Length', fileRes.headers['content-length']);
    fileRes.data.pipe(res);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});
